编写代码对flag文件进行解码

#include <stdio.h>
#include <stdlib.h>

int main()
{
    FILE *fileOld = fopen("D:\\flag.txt","rb");
    FILE *fileNew = fopen("D:\\flagNew.txt","rb+");
    char v1,v4,v5;
    v4 = 34;
    v5 = 0;
    while(1)
    {
        v1 = fgetc(fileOld);
        if( v1 == -1 )
            break;
        fputc(((v1^v4)-v5),fileNew);
        v4 += 34;
        v5 = (v5 + 2) & 0xF;
    }
    fclose(fileNew);
    fclose(fileOld);
    return 0;
}
